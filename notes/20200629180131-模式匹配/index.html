模式匹配

June 29, 2020 · 1 min · Gray King 
#[derive(Debug)]
enum UsState {
	Alabama,
	Alaska,
}

enum Coin {
	Penny,
	Nickel,
	Dime,
	Quarter(UsState),
}

fn value_cents(coin: Coin) -> u8 {

	match coin {
		Coin::Peny => {
			println!("Lucy penny!");
			1
		},
		Coin::Nickel => 5,
		Coin::Dime => 10,
		Coin::Quarter(state) => {
			println!("State quarter from {:?}!", state);
			25
		},
	}
}

// Option
fn plus_one(x: Option<i32>) -> Option<i32> {
	match x {
		None => None,
		Some(i) => Some(i + 1),
	}
}

let five = Some(5);
let six = plus_one(five);
let none = plus_one(None);

编译器会检查 match 覆盖所有情况， _ 可以用于所有未覆盖的情况。
