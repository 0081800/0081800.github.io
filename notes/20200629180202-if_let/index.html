if let

<span title='2020-06-29 18:02:00 +0800 +0800'>June 29, 2020</span>&nbsp;·&nbsp;1 min&nbsp;·&nbsp;Gray King 
let some_u8_value = Some(0u8);

// Without `if let`
match some_u8_value {
	Some(3) => println!("three");
	- => (),
}

// With `if let`
if let Some(3) = some_u8_value {
	println!("three");
}
