xinetd 代理 SMTP 和 IMAP#

通过 xinetd 代理 SMTP 和 IMAP 实现 gmail 翻墙。

  1. 配置服务端

    service imap
{
		type = UNLISTED
		port = 993
		bind = 0.0.0.0
		socket_type = stream
		wait = no
		user = nobody
		redirect = imap.gmail.com 993
		per_source = UNLIMITED
		cps = 100 2
}
service smtp-465
{
		type = UNLISTED
		port = 465
		bind = 0.0.0.0
		socket_type = stream
		wait = no
		user = nobody
		redirect = smtp.gmail.com 465
		per_source = UNLIMITED
		cps = 100 2
}
service smtp-587
{
		type = UNLISTED
		port = 587
		bind = 0.0.0.0
		socket_type = stream
		wait = no
		user = nobody
		redirect = smtp.gmail.com 587
		per_source = UNLIMITED
		cps = 100 2
}

  2. 本机修改 hosts

    echo '45.115.36.35 smtp.gmail.com' | sudo tee -a /etc/hosts
echo '45.115.36.35 imap.gmail.com' | sudo tee -a /etc/hosts
